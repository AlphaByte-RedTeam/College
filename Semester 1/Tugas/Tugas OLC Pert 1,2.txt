    cout << "Hasilnya adalah = " <<f <<endl;

Bagi mahasiswa yang mau menjawab soal nomor 5, jawaban atau diskusinya bisa di thread ini. 

Ulangi soal nomor 4 untuk konversi temperature dari Fahrenheit ke Celcius, dengan rumus C = 5 / 9 x (F - 32)

Kode :

#include <iostream>
using namespace std;

int main(){

    cout << "Masukkan Suhu Dalam Fahrenheit : "; //input Fahrenheit
    double f;
    cin >> f;
    
    double c;
    c = (f - 32) * 5/9;
    cout <<"Hasilnya adalah = " <<c <<"C";
    
    return 0;
}
